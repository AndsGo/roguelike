创建 `src/managers/DailyChallengeManager.ts`：

```typescript
import { SeededRNG } from '../utils/rng';
import { SaveManager } from './SaveManager';

export interface DailyModifiers {
  title: string;
  description: string;
  seed: number;
  rules: DailyRule[];
  difficulty: string;
}

export interface DailyRule {
  type: 'enemy_element' | 'hero_restriction' | 'gold_modifier' | 'hp_modifier';
  label: string;
  value: any;
}

// 每日挑战规则池
const RULE_POOL: Array<(rng: SeededRNG) => DailyRule> = [
  (rng) => {
    const elements = ['fire', 'ice', 'lightning', 'dark', 'holy'] as const;
    const el = elements[rng.nextInt(0, elements.length - 1)];
    const names: Record<string, string> = { fire: '火', ice: '冰', lightning: '雷', dark: '暗', holy: '圣' };
    return { type: 'enemy_element', label: `敌人强化: ${names[el]}属性+20%`, value: el };
  },
  (_rng) => ({ type: 'gold_modifier', label: '金币获取-30%', value: 0.7 }),
  (_rng) => ({ type: 'gold_modifier', label: '金币获取+50%', value: 1.5 }),
  (_rng) => ({ type: 'hp_modifier', label: '英雄初始HP-20%', value: 0.8 }),
];

const DAILY_TITLES = [
  '烈焰试炼', '寒冰考验', '暗影挑战', '雷霆之路', '圣光征途',
  '绝地求生', '速通之日', '铁壁防线',
];

export class DailyChallengeManager {
  private static STORAGE_KEY = 'roguelike_daily';

  static generateSeedFromDate(dateStr: string): number {
    let hash = 0;
    for (let i = 0; i < dateStr.length; i++) {
      const char = dateStr.charCodeAt(i);
      hash = ((hash << 5) - hash) + char;
      hash = hash & hash; // Convert to 32bit integer
    }
    return Math.abs(hash ^ 0xdeadbeef);
  }

  static getTodayString(): string {
    const now = new Date();
    const y = now.getFullYear();
    const m = String(now.getMonth() + 1).padStart(2, '0');
    const d = String(now.getDate()).padStart(2, '0');
    return `${y}${m}${d}`;
  }

  static getTodaysSeed(): number {
    return this.generateSeedFromDate(this.getTodayString());
  }

  static getDailyModifiers(seed: number): DailyModifiers {
    const rng = new SeededRNG(seed);

    // Pick title
    const title = DAILY_TITLES[rng.nextInt(0, DAILY_TITLES.length - 1)];

    // Pick 1-2 rules
    const ruleCount = rng.nextInt(1, 2);
    const rules: DailyRule[] = [];
    const usedIndices = new Set<number>();
    for (let i = 0; i < ruleCount; i++) {
      let idx: number;
      do { idx = rng.nextInt(0, RULE_POOL.length - 1); } while (usedIndices.has(idx));
      usedIndices.add(idx);
      rules.push(RULE_POOL[idx](rng));
    }

    // Pick difficulty (hard or nightmare)
    const difficulties = ['hard', 'nightmare'];
    const difficulty = difficulties[rng.nextInt(0, difficulties.length - 1)];

    const description = `难度: ${difficulty === 'hard' ? '困难' : '噩梦'} | ${rules.map(r => r.label).join(' | ')}`;

    return { title, description, seed, rules, difficulty };
  }

  static isCompletedToday(): boolean {
    const data = SaveManager.loadData<{ date: string; completed: boolean }>(this.STORAGE_KEY);
    return data?.date === this.getTodayString() && data?.completed === true;
  }

  static markCompleted(): void {
    SaveManager.saveData(this.STORAGE_KEY, {
      date: this.getTodayString(),
      completed: true,
    });
  }

  static getBestScore(): number {
    const data = SaveManager.loadData<{ bestScore: number }>('roguelike_daily_best');
    return data?.bestScore ?? 0;
  }

  static updateBestScore(score: number): void {
    if (score > this.getBestScore()) {
      SaveManager.saveData('roguelike_daily_best', { bestScore: score });
    }
  }
}
```

```typescript
static applyGoldModifier(baseGold: number, rules: DailyRule[]): number {
  const goldRule = rules.find(r => r.type === 'gold_modifier');
  return goldRule ? Math.round(baseGold * goldRule.value) : baseGold;
}

static applyHPModifier(baseHP: number, rules: DailyRule[]): number {
  const hpRule = rules.find(r => r.type === 'hp_modifier');
  return hpRule ? Math.round(baseHP * hpRule.value) : baseHP;
}

static getEnemyElementBonus(rules: DailyRule[]): string | null {
  const elRule = rules.find(r => r.type === 'enemy_element');
  return elRule ? elRule.value : null;
}
```

**Step 2: 实现 BuildReviewPanel**

创建 `src/ui/BuildReviewPanel.ts`：

```typescript
import Phaser from 'phaser';
import { Panel } from './Panel';
import { Theme } from './Theme';
import { Button } from './Button';
import { RunStats } from '../types';
import { UI } from '../i18n';

export class BuildReviewPanel {
  private backdrop: Phaser.GameObjects.Rectangle;
  private panel: Panel;
  private closeBtn: Button;
  private scene: Phaser.Scene;

  constructor(
    scene: Phaser.Scene,
    runStats: RunStats,
    heroStates: any[], // HeroState[]
    relicIds: string[],
    synergies: any[],
  ) {
    this.scene = scene;
    const W = 600, H = 380;
    const cx = 400, cy = 225;

    // Backdrop
    this.backdrop = scene.add.rectangle(cx, cy, 800, 450, 0x000000, 0.6)
      .setDepth(799).setInteractive();

    // Panel
    this.panel = new Panel(scene, cx - W/2, cy - H/2, W, H);
    this.panel.setDepth(800);

    // Close button
    this.closeBtn = new Button(scene, cx, cy + H/2 - 30, UI.common?.close ?? '[关闭]', 100, 32, () => {
      this.close();
    });
    this.closeBtn.setDepth(801);

    // Render content
    this.renderStats(runStats);
    this.renderHeroes(heroStates);
    this.renderRelics(relicIds);
    this.renderSynergies(synergies);
  }

  private renderStats(stats: RunStats): void {
    const lines: string[] = [];
    lines.push(`总伤害: ${stats.totalDamage}`);
    lines.push(`总治疗: ${stats.totalHealing}`);
    lines.push(`击杀数: ${stats.totalKills}`);
    lines.push(`暴击次数: ${stats.criticalHits ?? 0}`);
    lines.push(`最大连击: ${stats.maxCombo}`);
    lines.push(`使用技能: ${stats.skillsUsed}`);

    // Element reactions
    const reactions = stats.elementReactions ?? {};
    const reactionCount = Object.values(reactions).reduce((a, b) => a + b, 0);
    lines.push(`元素反应: ${reactionCount}次`);

    // Duration
    const durationSec = Math.round((stats.runDurationMs ?? 0) / 1000);
    const min = Math.floor(durationSec / 60);
    const sec = durationSec % 60;
    lines.push(`用时: ${min}分${sec}秒`);

    // Gold
    lines.push(`获得金币: ${stats.goldEarned}`);
    lines.push(`花费金币: ${stats.goldSpent}`);

    // Per-hero stats
    for (const [heroId, hs] of Object.entries(stats.heroStats ?? {})) {
      lines.push(`[${heroId}] 伤害:${hs.damage} 治疗:${hs.healing} 击杀:${hs.kills}`);
    }

    this.panel.addText(lines.join('\n'));
  }

  private renderHeroes(heroStates: any[]): void {
    // Display hero equipment summary
    for (const hero of heroStates) {
      const eqNames = (hero.equipment ?? [])
        .filter((e: any) => e != null)
        .map((e: any) => e.name ?? e.id)
        .join(', ');
      if (eqNames) {
        this.panel.addText(`${hero.heroId}: ${eqNames}`);
      }
    }
  }

  private renderRelics(relicIds: string[]): void {
    if (relicIds.length > 0) {
      this.panel.addText(`遗物 (${relicIds.length}): ${relicIds.join(', ')}`);
    }
  }

  private renderSynergies(synergies: any[]): void {
    if (synergies.length > 0) {
      const synStr = synergies
        .filter((s: any) => s.active)
        .map((s: any) => `${s.name}(${s.count})`)
        .join(', ');
      if (synStr) {
        this.panel.addText(`羁绊: ${synStr}`);
      }
    }
  }

  close(): void {
    this.backdrop.destroy();
    this.panel.destroy();
    this.closeBtn.destroy();
  }
}
```
