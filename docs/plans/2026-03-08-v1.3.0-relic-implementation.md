```typescript
import { RelicState, RelicEffect, UnitStats } from '../types';
import relicsData from '../data/relics.json';
import { EventBus } from './EventBus';

interface RelicDef {
  id: string;
  name: string;
  description: string;
  rarity: string;
  triggerEvent: string;
  effect: RelicEffect;
}

/**
 * Singleton evaluating relic effects during battle.
 * Three layers: stat boosts (passive), reactive (EventBus), formula mods (inline).
 */
export class RelicSystem {
  private static instance: RelicSystem | null = null;
  private relics: RelicState[] = [];
  private relicDefs: Map<string, RelicDef> = new Map();

  // Runtime state for kill_momentum etc.
  private runtimeBonuses: Map<string, Partial<Record<keyof UnitStats, number>>> = new Map();

  // Listener references for cleanup
  private listeners: Array<{ event: string; handler: (...args: any[]) => void }> = [];

  private constructor() {
    for (const r of relicsData as RelicDef[]) {
      this.relicDefs.set(r.id, r);
    }
  }

  private static getInstance(): RelicSystem {
    if (!RelicSystem.instance) {
      RelicSystem.instance = new RelicSystem();
    }
    return RelicSystem.instance;
  }

  /** Activate relic system for a battle with given held relics */
  static activate(relics: RelicState[]): void {
    const inst = RelicSystem.getInstance();
    inst.relics = relics;
    inst.runtimeBonuses.clear();
    inst.registerListeners();
  }

  /** Deactivate and clean up all listeners */
  static deactivate(): void {
    const inst = RelicSystem.getInstance();
    inst.unregisterListeners();
    inst.runtimeBonuses.clear();
  }

  /** Reset for testing */
  static reset(): void {
    if (RelicSystem.instance) {
      RelicSystem.instance.unregisterListeners();
    }
    RelicSystem.instance = null;
  }

  /** Get flat stat modifiers from all passive/stat_boost relics */
  static getStatModifiers(): Partial<Record<keyof UnitStats, number>> {
    const inst = RelicSystem.getInstance();
    const mods: Partial<Record<keyof UnitStats, number>> = {};

    for (const relic of inst.relics) {
      const def = inst.relicDefs.get(relic.id);
      if (!def) continue;

      if (def.effect.type === 'stat_boost' && def.effect.stat && def.effect.value != null) {
        const key = def.effect.stat as keyof UnitStats;
        mods[key] = ((mods[key] as number) ?? 0) + def.effect.value;
      }
    }

    return mods;
  }

  /** Check if a specific relic is held */
  static hasRelic(relicId: string): boolean {
    return RelicSystem.getInstance().relics.some(r => r.id === relicId);
  }

  /** Get the definition for a relic */
  static getRelicDef(relicId: string): RelicDef | undefined {
    return RelicSystem.getInstance().relicDefs.get(relicId);
  }

  private registerListeners(): void {
    // Will be implemented in Task 2
  }

  private unregisterListeners(): void {
    const bus = EventBus.getInstance();
    for (const { event, handler } of this.listeners) {
      bus.off(event as any, handler);
    }
    this.listeners = [];
  }
}
```

**Step 2: Implement reactive listener registration**

In `src/systems/RelicSystem.ts`, implement `registerListeners()`:

```typescript
private registerListeners(): void {
  this.unregisterListeners(); // Clear any existing
  const bus = EventBus.getInstance();

  for (const relic of this.relics) {
    const def = this.relicDefs.get(relic.id);
    if (!def) continue;

    // Skip stat_boost and passive types — handled in getStatModifiers/getFormulaModifiers
    if (def.effect.type === 'stat_boost' || def.effect.type === 'formula_mod') continue;

    const handler = this.createHandler(relic, def);
    if (handler) {
      bus.on(def.triggerEvent as any, handler);
      this.listeners.push({ event: def.triggerEvent, handler });
    }
  }
}

private createHandler(relic: RelicState, def: RelicDef): ((data: any) => void) | null {
  const effect = def.effect;

  switch (effect.type) {
    case 'on_battle_start':
      return (data: any) => this.handleBattleStart(relic, def, data);
    case 'on_damage':
      return (data: any) => this.handleOnDamage(relic, def, data);
    case 'on_kill':
      return (data: any) => this.handleOnKill(relic, def, data);
    case 'on_heal':
      return (data: any) => this.handleOnHeal(relic, def, data);
    case 'on_battle_end':
      return (data: any) => this.handleBattleEnd(relic, def, data);
    default:
      return null;
  }
}
```

Implement each handler stub. The detailed handler logic:

```typescript
private heroes: Unit[] = [];
private enemies: Unit[] = [];

static activateWithUnits(relics: RelicState[], heroes: Unit[], enemies: Unit[]): void {
  const inst = RelicSystem.getInstance();
  inst.relics = relics;
  inst.heroes = heroes;
  inst.enemies = enemies;
  inst.runtimeBonuses.clear();
  inst.registerListeners();
}

private handleBattleStart(relic: RelicState, def: RelicDef, _data: any): void {
  const val = def.effect.value ?? 0;

  switch (def.id) {
    case 'iron_heart':
      // +50 shield — add temporary HP
      for (const hero of this.heroes) {
        hero.currentHp = Math.min(hero.currentHp + val, hero.currentStats.maxHp + val);
      }
      break;
    case 'herb_pouch':
    case 'life_spring':
      // Heal % of max HP
      for (const hero of this.heroes) {
        const heal = Math.round(hero.currentStats.maxHp * val);
        hero.heal(heal);
      }
      break;
    case 'time_crystal':
      // Reduce all skill cooldowns by 20%
      for (const hero of this.heroes) {
        for (const [skillId, cd] of hero.skillCooldowns) {
          hero.skillCooldowns.set(skillId, cd * (1 - val));
        }
      }
      break;
  }
  relic.triggerCount++;
}

private handleOnDamage(relic: RelicState, def: RelicDef, data: any): void {
  const val = def.effect.value ?? 0;
  const chance = def.effect.chance ?? 1;

  // Roll chance
  if (chance < 1 && Math.random() > chance) return;

  const source = this.findUnit(data.sourceId);
  const target = this.findUnit(data.targetId);
  if (!source || !target) return;

  switch (def.id) {
    case 'vampiric_fang':
      // 3% lifesteal
      if (source.isHero) {
        const healAmt = Math.round(data.amount * val);
        if (healAmt > 0) source.heal(healAmt);
      }
      break;
    case 'soul_mirror':
      // Reflect 15% damage back
      if (target.isHero && source.isAlive) {
        const reflect = Math.round(data.amount * val);
        if (reflect > 0) source.takeDamage(reflect);
      }
      break;
    case 'thunder_emblem':
      // 10% chance extra lightning damage
      if (source.isHero) {
        const nearby = this.enemies.filter(e => e.isAlive && e.unitId !== target.unitId);
        if (nearby.length > 0) {
          nearby[0].takeDamage(val);
        }
      }
      break;
  }
  relic.triggerCount++;
}

private handleOnKill(relic: RelicState, def: RelicDef, data: any): void {
  const val = def.effect.value ?? 0;
  const killer = this.findUnit(data.killerId);
  if (!killer || !killer.isHero) return;

  switch (def.id) {
    case 'blood_vial':
      // Heal 5% max HP on kill
      const healAmt = Math.round(killer.currentStats.maxHp * val);
      killer.heal(healAmt);
      break;
    case 'soul_collector':
      // +1 attack permanently (this battle)
      killer.currentStats.attack += val;
      break;
  }
  relic.triggerCount++;
}

private handleOnHeal(relic: RelicState, def: RelicDef, data: any): void {
  // healers_blessing: +20% healing — handled in DamageSystem (Task 3)
  relic.triggerCount++;
}

private handleBattleEnd(relic: RelicState, _def: RelicDef, _data: any): void {
  // lucky_coin, training_manual, tactics_manual — gold/exp bonuses
  // These are applied in BattleScene.handleBattleEnd (Task 4)
  relic.triggerCount++;
}

private findUnit(unitId: string): Unit | undefined {
  return [...this.heroes, ...this.enemies].find(u => u.unitId === unitId);
}
```

```typescript
    if (relic.id === 'armor_piercer') total += 0.3;

  return total;

/** Get total heal bonus multiplier (e.g. 0.2 = +20%) */
static getHealBonus(): number {
  const inst = RelicSystem.getInstance();

  for (const relic of inst.relics) {
    const def = inst.relicDefs.get(relic.id);
    if (!def) continue;
    if (def.id === 'healers_blessing' && def.effect.value) {
      total += def.effect.value;

  return total;
```

```typescript
/** mono_element_crown: +40% if all hero elements match */
static getConditionalDamageBonus(heroElements: (string | undefined)[]): number {
  if (!RelicSystem.hasRelic('mono_element_crown')) return 0;
  const elements = heroElements.filter(Boolean);
  if (elements.length === 0) return 0;
  const allSame = elements.every(e => e === elements[0]);
  return allSame ? 0.4 : 0;
}

/** diversity_badge: +8% attackSpeed per unique class */
static getAttackSpeedBonus(uniqueClassCount: number): number {
  if (!RelicSystem.hasRelic('diversity_badge')) return 0;
  return uniqueClassCount * 0.08;
}

/** berserker_oath: +20% attack +10% crit when no healer */
static getBerserkerBonus(hasHealer: boolean): { attack: number; critChance: number } {
  if (!RelicSystem.hasRelic('berserker_oath') || hasHealer) {
    return { attack: 0, critChance: 0 };
```

```typescript
/** Get combined conditional stat modifiers based on team composition */
static getConditionalStatMods(): Partial<Record<keyof UnitStats, number>> {
  const inst = RelicSystem.getInstance();
  const mods: Partial<Record<keyof UnitStats, number>> = {};

  if (inst.heroes.length === 0) return mods;

  // mono_element_crown
  const heroElements = inst.heroes.map(h => h.element);
  const monoBonus = RelicSystem.getConditionalDamageBonus(heroElements);
  // monoBonus applied as damage multiplier in DamageSystem, not as stat mod

  // diversity_badge
  const uniqueClasses = new Set(inst.heroes.map(h => h.classType)).size;
  const speedBonus = RelicSystem.getAttackSpeedBonus(uniqueClasses);
  if (speedBonus > 0) {
    mods.attackSpeed = speedBonus;
  }

  // berserker_oath
  const hasHealer = inst.heroes.some(h => h.role === 'healer');
  const berserk = RelicSystem.getBerserkerBonus(hasHealer);
  if (berserk.attack > 0) {
    // These are percentage-based, applied as flat for simplicity
    // Average hero attack ~80, 20% = +16
    // Better: store as multiplier flag, apply in DamageSystem
    mods.critChance = (mods.critChance ?? 0) + berserk.critChance;
  }

  return mods;
}
```

```typescript
applyHeal(healer: Unit, target: Unit, baseHeal: number): number {
  let healAmount = baseHeal;
  const healBonus = RelicSystem.getHealBonus();
  if (healBonus > 0) {
    healAmount = Math.round(healAmount * (1 + healBonus));
  }

  const actual = target.heal(healAmount);

  // Overflow shield: excess healing becomes temporary HP
  if (RelicSystem.hasOverflowShield() && target.isHero) {
    const overflow = healAmount - actual;
    if (overflow > 0) {
      const maxShield = Math.round(target.currentStats.maxHp * 0.2);
      const shieldAmount = Math.min(overflow, maxShield);
      // Add as temporary maxHp + currentHp boost
      target.currentHp += shieldAmount;
      // Don't increase maxHp — just let HP exceed max temporarily
    }
  }

  if (actual > 0) {
    new DamageNumber(target.scene, target.x + this.rng.nextInt(-10, 10), target.y - 20, actual, true);
  }
  return actual;
}
```

```typescript
  fire_emblem: 'fire',
  ice_crystal_pendant: 'ice',
  lightning_rod: 'lightning',
  dark_grimoire: 'dark',

/** Get element-specific damage bonus for a given element */
static getElementDamageBonus(element: string): number {
  const inst = RelicSystem.getInstance();
  let total = 0;
  for (const relic of inst.relics) {
    // Element-specific relics
    const relicElement = RelicSystem.ELEMENT_RELIC_MAP[relic.id];
    if (relicElement === element) {
      const def = inst.relicDefs.get(relic.id);
      total += def?.effect.value ?? 0;
    }
    // Crown of elements: all elements
    if (relic.id === 'crown_of_elements') {
      const def = inst.relicDefs.get(relic.id);
      total += def?.effect.value ?? 0;
    }
  }
  return total;
}

/** berserker_mask: +25% attack when HP ratio < 50% */
static getLowHpAttackBonus(hpRatio: number): number {
```
