# Formation System Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Add a front/back row formation system where hero row placement affects enemy melee targeting priority.

**Architecture:** HeroState gains a `formation: 'front' | 'back'` field with role-based defaults. BattleScene positions heroes at two X values by row. TargetingSystem adds a front-row aggro bonus for melee enemies. MapScene gets a modal FormationPanel for drag/tap row reassignment.

**Tech Stack:** TypeScript, Phaser 3, Vitest

**Step 1: Add getHeroFormation helper to RunManager**

```typescript
/** Get hero's formation, falling back to role-based default for old saves */
getHeroFormation(heroId: string): 'front' | 'back' {
  const hero = this.state.heroes.find(h => h.id === heroId);
  if (!hero) return 'front';
  return hero.formation ?? autoFormationByRole(this.getHeroData(heroId).role);
}

/** Update a hero's formation row */
setHeroFormation(heroId: string, formation: 'front' | 'back'): void {
  const hero = this.state.heroes.find(h => h.id === heroId);
  if (hero) hero.formation = formation;
}
```

### Task 3: Add front-row targeting bonus to TargetingSystem

**Files:**
- Modify: `src/systems/TargetingSystem.ts:224-242` — add formation bonus in scoring loop
- Test: `tests/systems/formation-targeting.test.ts`

**Context:** The `selectWithElementWeight()` method (line 171-246) does a two-pass scoring. In the second pass (line 224-243), we add a front-row bonus for melee enemy attackers. The bonus only applies when the attacking unit is a melee enemy (attackRange <= MELEE_RANGE_THRESHOLD).

**Step 1: Add formation scoring**

In `src/systems/TargetingSystem.ts`, import the constants:
```typescript
import { FRONT_ROW_AGGRO_BONUS, MELEE_RANGE_THRESHOLD } from '../constants';
```

In `selectWithElementWeight()`, in the second-pass loop (line ~224-243), after the aggro weight line (`score += rawThreat[i]...`), add:

```typescript
      // Formation bonus: melee enemies prefer front-row heroes
      if (!unit.isHero && unit.currentStats.attackRange <= MELEE_RANGE_THRESHOLD) {
        if (targets[i].formation === 'front') {
          score += FRONT_ROW_AGGRO_BONUS * invMaxBase;
        }
      }
```

The `invMaxBase` normalization keeps the bonus proportional to other scoring factors. With FRONT_ROW_AGGRO_BONUS=50, this creates a strong but not absolute preference for front-row targets.

```typescript
import { describe, it, expect, beforeEach } from 'vitest';
import { TargetingSystem } from '../../src/systems/TargetingSystem';
import { Unit } from '../../src/entities/Unit';
import { createScene } from '../helpers/scene-harness';

function createMockUnit(
  scene: any,
  id: string,
  isHero: boolean,
  formation: 'front' | 'back' = 'front',
  attackRange = 35,
): Unit {
  const stats = {
    maxHp: 100, hp: 100, attack: 10, defense: 5,
    magicPower: 5, magicResist: 5, speed: 100,
    attackSpeed: 1, attackRange, critChance: 0, critDamage: 1.5,
  };
  const unit = new Unit(scene, 100, 200, id, id, 'melee_dps', stats, isHero, 0xff0000, 0x880000);
  unit.formation = formation;
  return unit;
}

describe('Formation targeting bonus', () => {
  let scene: any;

  beforeEach(() => {
    scene = createScene();
    TargetingSystem.resetThreat();
    TargetingSystem.beginFrame(1000); // ensure fresh frame
  });

  it('melee enemy prefers front-row hero over back-row', () => {
    const enemy = createMockUnit(scene, 'enemy1', false, 'front', 35); // melee
    const heroFront = createMockUnit(scene, 'hero1', true, 'front');
    const heroBack = createMockUnit(scene, 'hero2', true, 'back');
    // Place them at equal distance
    heroFront.x = 200; heroFront.y = 200;
    heroBack.x = 200; heroBack.y = 200;
    enemy.x = 400; enemy.y = 200;

    const target = TargetingSystem.selectTarget(enemy, [heroFront, heroBack], []);
    expect(target?.unitId).toBe('hero1');
  });

  it('ranged enemy does NOT get front-row bonus', () => {
    const enemy = createMockUnit(scene, 'enemy1', false, 'front', 200); // ranged
    const heroFront = createMockUnit(scene, 'hero1', true, 'front');
    const heroBack = createMockUnit(scene, 'hero2', true, 'back');
    // Place back-row hero closer to enemy
    heroFront.x = 200; heroFront.y = 200;
    heroBack.x = 300; heroBack.y = 200;
    enemy.x = 400; enemy.y = 200;

    const target = TargetingSystem.selectTarget(enemy, [heroFront, heroBack], []);
    // Ranged enemy targets nearest (heroBack is closer), no formation bias
    expect(target?.unitId).toBe('hero2');
  });

  it('hero targeting enemy is not affected by formation bonus', () => {
    const hero = createMockUnit(scene, 'hero1', true, 'front', 35);
    const enemyFront = createMockUnit(scene, 'enemy1', false, 'front');
    const enemyBack = createMockUnit(scene, 'enemy2', false, 'back');
    // Place both enemies at equal distance
    enemyFront.x = 400; enemyFront.y = 200;
    enemyBack.x = 400; enemyBack.y = 200;
    hero.x = 200; hero.y = 200;

    // Hero should pick based on normal scoring, not formation
    const target = TargetingSystem.selectTarget(hero, [enemyFront, enemyBack], []);
    expect(target).toBeTruthy(); // just verify no crash, either is valid
  });
});
```

**Files:**

- Modify: `src/i18n.ts` — (strings already added in Task 1)

**Context:** The MapScene has a hero panel at the bottom (y = GAME_HEIGHT - 70). We add a small "阵型" button near it. Clicking opens a FormationPanel modal. The panel uses the standard modal pattern (backdrop d799, panel d800, close d801). Heroes are displayed as chibi sprites and can be **tapped to toggle** between front and back row (simpler than drag, consistent with the 800x450 canvas).

**Step 1: Create FormationPanel**

Create `src/ui/FormationPanel.ts`:

```typescript
import Phaser from 'phaser';
import { GAME_WIDTH, GAME_HEIGHT } from '../constants';
import { Theme, colorToString } from './Theme';
import { Button } from './Button';
import { HeroState, HeroData } from '../types';
import { RunManager } from '../managers/RunManager';
import { autoFormationByRole } from '../managers/RunManager';
import { getOrCreateTexture, ChibiConfig } from '../systems/UnitRenderer';
import { UI } from '../i18n';

const PANEL_W = 400;
const PANEL_H = 280;

export class FormationPanel extends Phaser.GameObjects.Container {
  private backdrop: Phaser.GameObjects.Rectangle;
  private heroSprites: Map<string, Phaser.GameObjects.Container> = new Map();

  constructor(scene: Phaser.Scene) {
    super(scene, 0, 0);
    scene.add.existing(this);
    this.setDepth(800);

    // Backdrop
    this.backdrop = scene.add.rectangle(GAME_WIDTH / 2, GAME_HEIGHT / 2, GAME_WIDTH, GAME_HEIGHT, 0x000000, 0.5)
      .setDepth(799).setInteractive();
    this.backdrop.on('pointerup', (p: Phaser.Input.Pointer) => {
      const px = p.x, py = p.y;
      const left = (GAME_WIDTH - PANEL_W) / 2;
      const top = (GAME_HEIGHT - PANEL_H) / 2;
      if (px < left || px > left + PANEL_W || py < top || py > top + PANEL_H) {
        this.close();
      }
    });

    // Panel background
    const panelX = (GAME_WIDTH - PANEL_W) / 2;
    const panelY = (GAME_HEIGHT - PANEL_H) / 2;
    const bg = scene.add.graphics().setDepth(800);
    bg.fillStyle(Theme.colors.panel, 0.95);
    bg.fillRoundedRect(panelX, panelY, PANEL_W, PANEL_H, 8);
    bg.lineStyle(2, Theme.colors.panelBorder, 1);
    bg.strokeRoundedRect(panelX, panelY, PANEL_W, PANEL_H, 8);
    this.add(bg);

    // Title
    const title = scene.add.text(GAME_WIDTH / 2, panelY + 20, UI.formation.title, {
      fontSize: '14px', color: '#ffffff', fontFamily: 'monospace', fontStyle: 'bold',
    }).setOrigin(0.5).setDepth(800);
    this.add(title);

    // Tip
    const tip = scene.add.text(GAME_WIDTH / 2, panelY + 38, UI.formation.tip, {
      fontSize: '9px', color: '#aaaaaa', fontFamily: 'monospace',
    }).setOrigin(0.5).setDepth(800);
    this.add(tip);

    // Column labels
    const frontX = GAME_WIDTH / 2 + 60;
    const backX = GAME_WIDTH / 2 - 60;
    scene.add.text(frontX, panelY + 58, UI.formation.front, {
      fontSize: '11px', color: '#ff8844', fontFamily: 'monospace', fontStyle: 'bold',
    }).setOrigin(0.5).setDepth(800);
    scene.add.text(backX, panelY + 58, UI.formation.back, {
      fontSize: '11px', color: '#44aaff', fontFamily: 'monospace', fontStyle: 'bold',
    }).setOrigin(0.5).setDepth(800);

    // Divider line
    const divider = scene.add.graphics().setDepth(800);
    divider.lineStyle(1, 0x555555, 0.5);
    divider.lineBetween(GAME_WIDTH / 2, panelY + 70, GAME_WIDTH / 2, panelY + PANEL_H - 50);
    this.add(divider);

    // Draw heroes
    this.refreshHeroes();

    // Auto-assign button
    new Button(scene, GAME_WIDTH / 2, panelY + PANEL_H - 28, UI.formation.autoAssign, () => {
      this.autoAssignAll();
    }, { width: 100, height: 24, fontSize: '9px', depth: 801 });

    // Close button
    new Button(scene, panelX + PANEL_W - 20, panelY + 12, '✕', () => {
      this.close();
    }, { width: 24, height: 24, fontSize: '12px', depth: 801 });
  }

  private refreshHeroes(): void {
    // Clear existing hero sprites
    for (const container of this.heroSprites.values()) {
      container.destroy();
    }
    this.heroSprites.clear();

    const rm = RunManager.getInstance();
    const heroes = rm.getHeroes();
    const panelY = (GAME_HEIGHT - PANEL_H) / 2;
    const frontX = GAME_WIDTH / 2 + 60;
    const backX = GAME_WIDTH / 2 - 60;

    const frontHeroes = heroes.filter(h => rm.getHeroFormation(h.id) === 'front');
    const backHeroes = heroes.filter(h => rm.getHeroFormation(h.id) === 'back');

    const drawRow = (list: HeroState[], x: number) => {
      const startY = panelY + 80;
      list.forEach((hero, i) => {
        const y = startY + i * 55;
        const data = rm.getHeroData(hero.id);

        const container = this.scene.add.container(x, y).setDepth(800);

        // Chibi sprite
        const config: ChibiConfig = {
          role: data.role,
          race: (data.race ?? 'human') as any,
          classType: (data.class ?? 'warrior') as any,
          fillColor: Theme.colors.role[data.role] ?? 0x888888,
          borderColor: 0x000000,
          isHero: true,
          isBoss: false,
          element: data.element as any,
        };
        const textureKey = getOrCreateTexture(this.scene, config);
        const sprite = this.scene.add.image(0, 0, textureKey).setDepth(800);
        container.add(sprite);

        // Name
        const name = this.scene.add.text(0, 22, data.name, {
          fontSize: '9px', color: '#ffffff', fontFamily: 'monospace',
        }).setOrigin(0.5).setDepth(800);
        container.add(name);

        // Make interactive — tap to toggle
        const hitZone = this.scene.add.rectangle(0, 0, 60, 50, 0x000000, 0)
          .setInteractive({ useHandCursor: true }).setDepth(801);
        hitZone.on('pointerup', () => {
          const current = rm.getHeroFormation(hero.id);
          rm.setHeroFormation(hero.id, current === 'front' ? 'back' : 'front');
          this.refreshHeroes();
        });
        container.add(hitZone);

        this.heroSprites.set(hero.id, container);
      });
    };

    drawRow(frontHeroes, frontX);
    drawRow(backHeroes, backX);
  }

  private autoAssignAll(): void {
    const rm = RunManager.getInstance();
    const heroes = rm.getHeroes();
    for (const hero of heroes) {
      const data = rm.getHeroData(hero.id);
      rm.setHeroFormation(hero.id, autoFormationByRole(data.role));
    }
    this.refreshHeroes();
  }

  close(): void {
    this.backdrop.destroy();
    this.destroy();
  }
}
```

import { RunManager } from '../../src/managers/RunManager';
import { SaveManager } from '../../src/managers/SaveManager';

describe('Formation save/load', () => {
  it('formation persists through save/load cycle', () => {
    const rm = RunManager.getInstance();
    rm.newRun(12345, 'normal', ['warrior', 'archer']);

    // Verify default assignments
    expect(rm.getHeroFormation('warrior')).toBe('front'); // melee_dps -> front (warrior hero has role melee_dps or tank)
    expect(rm.getHeroFormation('archer')).toBe('back');  // ranged_dps -> back

    // Change warrior to back
    rm.setHeroFormation('warrior', 'back');
    expect(rm.getHeroFormation('warrior')).toBe('back');

    // Save and reload
    SaveManager.saveData(0);
    rm.newRun(99999, 'normal', ['warrior', 'archer']); // reset state
    SaveManager.loadData(0);

    // Should restore the changed formation
    expect(rm.getHeroFormation('warrior')).toBe('back');
  });

  it('old save without formation falls back to role default', () => {
    const rm = RunManager.getInstance();
    rm.newRun(12345, 'normal', ['warrior', 'archer']);

    // Simulate old save: remove formation field
    const heroes = rm.getHeroes();
    for (const h of heroes) {
      delete (h as any).formation;
    }

    // getHeroFormation should fall back to role-based default
    expect(rm.getHeroFormation('warrior')).toBeDefined();
    expect(rm.getHeroFormation('archer')).toBeDefined();
  });
});
