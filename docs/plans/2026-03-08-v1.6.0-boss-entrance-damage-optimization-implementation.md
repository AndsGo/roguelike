**Step 2: Implement DamageAccumulator**

Create `src/systems/DamageAccumulator.ts`:

```typescript
import { DamageNumber, DamageNumberConfig } from '../components/DamageNumber';
import { ElementType } from '../types';

const ACCUMULATE_WINDOW = 150; // ms — merge hits within this window
const MAX_VISIBLE_PER_UNIT = 3;
const VERTICAL_SPACING = 18; // px between stacked numbers
const BASE_Y_OFFSET = -20;

interface PendingEntry {
  scene: Phaser.Scene;
  x: number;
  y: number;
  damageTotal: number;
  healTotal: number;
  isCrit: boolean;
  element?: ElementType;
  comboCount: number;
  timer: number;
}

export class DamageAccumulator {
  private pending: Map<string, PendingEntry> = new Map();
  private activeSlots: Map<string, number[]> = new Map(); // unitId → active Y offsets with expiry

  addDamage(
    targetId: string,
    scene: Phaser.Scene,
    x: number,
    y: number,
    amount: number,
    config?: DamageNumberConfig,
  ): void {
    const entry = this.getOrCreateEntry(targetId, scene, x, y);
    entry.damageTotal += amount;
    if (config?.isCrit) entry.isCrit = true;
    if (config?.element && !entry.element) entry.element = config.element;
    if (config?.comboCount && config.comboCount > entry.comboCount) {
      entry.comboCount = config.comboCount;
    }
  }

  addHeal(
    targetId: string,
    scene: Phaser.Scene,
    x: number,
    y: number,
    amount: number,
  ): void {
    const key = `${targetId}_heal`;
    const entry = this.getOrCreateEntry(key, scene, x, y);
    entry.healTotal += amount;
  }

  update(delta: number): void {
    for (const [key, entry] of this.pending) {
      entry.timer -= delta;
      if (entry.timer <= 0) {
        this.flush(key, entry);
        this.pending.delete(key);
      }
    }
  }

  private getOrCreateEntry(key: string, scene: Phaser.Scene, x: number, y: number): PendingEntry {
    let entry = this.pending.get(key);
    if (!entry) {
      entry = {
        scene,
        x,
        y,
        damageTotal: 0,
        healTotal: 0,
        isCrit: false,
        comboCount: 0,
        timer: ACCUMULATE_WINDOW,
      };
      this.pending.set(key, entry);
    }
    return entry;
  }

  private flush(key: string, entry: PendingEntry): void {
    const unitId = key.replace('_heal', '');
    const yOffset = this.getNextYOffset(unitId);

    if (entry.damageTotal > 0) {
      new DamageNumber(
        entry.scene,
        entry.x,
        entry.y + BASE_Y_OFFSET + yOffset,
        Math.round(entry.damageTotal),
        false,
        entry.isCrit,
        entry.element,
        entry.comboCount > 0 ? entry.comboCount : undefined,
      );
    }

    if (entry.healTotal > 0) {
      new DamageNumber(
        entry.scene,
        entry.x,
        entry.y + BASE_Y_OFFSET + yOffset - (entry.damageTotal > 0 ? VERTICAL_SPACING : 0),
        Math.round(entry.healTotal),
        true,
        false,
      );
    }
  }

  private getNextYOffset(unitId: string): number {
    let slots = this.activeSlots.get(unitId);
    if (!slots) {
      slots = [];
      this.activeSlots.set(unitId, slots);
    }
    const offset = -slots.length * VERTICAL_SPACING;

    // Track this slot, auto-remove after DAMAGE_NUMBER_DURATION
    slots.push(Date.now());
    if (slots.length > MAX_VISIBLE_PER_UNIT) {
      slots.shift(); // drop oldest
    }

    // Clean old slots (older than 800ms)
    const now = Date.now();
    const filtered = slots.filter(t => now - t < 800);
    this.activeSlots.set(unitId, filtered);

    return offset;
  }

  reset(): void {
    this.pending.clear();
    this.activeSlots.clear();
  }
}
```
